function resid_three_period2(a3::Float64, a2::Float64, params::Models)

    # 
    # (cash on hand)
    coh = (1 + params.rent)*a2 + params.y2
    if coh - a3 > 0.0 # 
        mu2 = mu_CRRA(coh - a3, params.)
    else
        mu2 = 10000.0
    end

    # 
    mu3 = mu_CRRA((1 + params.rent)*a3 + params.y3, params.)

    # (residual)
    return params.*(1.0 + params.rent)*(mu3/mu2) - 1.0

end

function resid_three_period1(a2::Float64, a1::Float64, a2_nl::Vector{Float64}, params::Models)

    # 
    # (cash on hand)
    coh = (1 + params.rent)*a1 + params.y1
    if coh - a2 > 0.0 # 
        mu1 = mu_CRRA(coh - a2, params.)
    else
        mu1 = 10000.0
    end

    # 
    interp1 = LinearInterpolation(params.grid_a, a2_nl)
    cons = interp1(a1)

    # 
    mu2 = mu_CRRA(cons, params.)

    # (residual)
    return params.*(1.0 + params.rent)*(mu2/mu1) - 1.0

end
