function resid_three_period2(a3::Float64, a2::Float64, e2::Int64, params::Models)

    # 
    # (cash on hand)
    coh = (1 + params.rent)*a2 + params.y2*params.endow[e2]
    if coh - a3 > 0.0 # 
        mu2 = mu_CRRA(coh - a3, params.)
    else
        mu2 = 10000.0
    end

    # 
    mu3 = mu_CRRA((1 + params.rent)*a3 + params.y3, params.)

    # (residual)
    return params.*(1 + params.rent)*(mu3/mu2) - 1.0

end

function resid_three_period1(a2::Float64, a1::Float64, e1::Int64, a2_nl::Matrix{Float64}, params::Models)

    # 
    # (cash on hand)
    coh = a1 + params.y1*params.endow[e1]
    if coh - a2 > 0.0 # 
        mu1 = mu_CRRA(coh - a2, params.)
    else
        mu1 = 10000.0
    end

    mu2 = zeros(params.ny)
    for i = 1:params.ny
        # 
        interp1 = LinearInterpolation(params.grid_a, a2_nl[:, i], extrapolation_bc=Line())
        a3_approx = interp1(a2)
        cons = (1 + params.rent)*a2 + params.y2*endow(i) - a3_approx

        # 
        mu2[i] = mu_CRRA(cons, params.)
    end

    # (residual)
    exp_val = dot(params.tran[e1, :], mu2)
    return params.*(1 + params.rent)*(exp_val/mu1) - 1.0

end
